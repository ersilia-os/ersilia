[docs]    def python_version(self, py_format=False):
        vi = sys.version_info
        if py_format:
            return "py{0}{1}".format(vi.major, vi.minor)
        else:
            return "{0}.{1}".format(vi.major, vi.minor)

[docs]    def ersilia_version_from_path(self, path):
        static_version_file = "_clean_static_version.py"
        fn = os.path.join(path, "ersilia", static_version_file)
        if not os.path.exists(fn):
            fn = os.path.join(path, static_version_file)
        if not os.path.exists(fn):
            raise Exception
        with open(fn, "r") as f:
            text = f.read()
            ver = text.split('"')[1]
        return ver

[docs]    def server_docker_name(self, tag=None, as_tuple=False):
        if tag is None:
            tag = "{0}-{1}".format(
                self.ersilia_version(), self.python_version(py_format=True)
            )
        org = self.cfg.EXT.DOCKERHUB_ORG
        img = self.cfg.ENV.DOCKER.SERVER_BASE_IMAGE
        if as_tuple:
            return org, img, tag
        else:
            name = "{0}/{1}:{2}".format(img, org, tag)
            return name
