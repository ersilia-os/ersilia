    def encode():
        """Get UUID code (long identifier)"""
        if uuid is None:
            alphabet = ALPHABET.lower()
            for n in PATTERN:
                s += ["".join([random.choice(alphabet) for _ in range(n)])]
            return "-".join(s)
        else:
            return str(uuid.uuid4())
