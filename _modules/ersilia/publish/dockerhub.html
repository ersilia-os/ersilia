[docs]    def upload(self):
        self.build_image()
        cmd = "docker login --password {0} --username {1}".format(
            self.docker_pwd, self.docker_user
        )
        run_command(cmd)
        cmd = "docker push {0}/{1}:{2}".format(
            DOCKERHUB_ORG, self.model_id, DOCKERHUB_LATEST_TAG
        )
        run_command(cmd)
